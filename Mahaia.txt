package Mahaia;
public class Mahaia {
	private static Mahaia NireMahaia;
	private KartaZerrenda mKarta;

	private Mahaia() {
		mKarta = new KartaZerrenda();
	}

	public static Mahaia getMahaia() {
		if (NireMahaia == null) {
			NireMahaia = new Mahaia();
		}
		return NireMahaia;
	}

	public void gehituKarta(Karta Karta) {
		mKarta.gehituKarta(Karta);
	}

	public Karta itzuliKarta() {
		Karta k = mKarta.getKarta(0);
		return k;

	}

	public boolean botaBai(Karta pKarta)
	{
		boolean bota = mKarta.berdina(pKarta, mKarta.getKarta(0));
		return bota;
	}

	public void imprimatuAzkena() {
		mKarta.imprimatuAzkena();
		
	}
}
// JUnit
import static org.junit.Assert.*;
import org.junit.Before;
import org.junit.Test;

public class MahaiaTest {

    private Mahaia mahaia;

    @Before
    public void setUp() {
        mahaia = Mahaia.getMahaia();
        jokalari = new Jokalari(); // Crear una instancia de la clase Jokalari
    }

    @Test
    public void testGehituKarta() {
        Karta karta = new Karta(); // Karta bat sortzen dugu
        mahaia.gehituKarta(karta); //Karta mahaura batzen dugu

        // Jokalariak notatzen duen karta gordetzen dugu
        Karta kartaBota = jokalari.kartaBota();

        assertEquals(kartaBota, mahaia.itzuliKarta()); // kartaBota eta gehituKarta konparatzen dugu
    }
}

    @Test
    public void testItzuliKarta() {
        // AsegurÃ©monos de que no haya una carta inicialmente
        assertNull(mahaia.itzuliKarta());

        // Karta bat gehitu
        Karta karta = new Karta();
        mahaia.gehituKarta(karta);

        // Ikusiko dugu batutako karta eta duguna berdina dela
        assertEquals(karta, mahaia.itzuliKarta());
    }
}
